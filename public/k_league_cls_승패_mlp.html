def team_encoding(train):
    train['홈 승리'] = train['홈 팀 결과'].apply(lambda x: 1 if x=='승' else 0) # 홈 승리 열 추가, 승리인 경우 1, 아닌 경우 0
    dic = {}
    # 각 홈팀별 이긴 경기 수를 딕셔너리에 저장
    for team in train['홈 팀 이름'].unique():
        value = train[train['홈 팀 이름'] == team]['홈 승리'].sum()
        #홈 팀 이름  열에서 고유한 팀 이름을 가져와 각 팀이 홈에서 이긴 경기 수 계산, 이 값을 dic에 저장
        dic[team] = value

    label_dic={}
    # 승리 횟수를 기준으로 오름차순 정렬, 각 팀에 대해 라벨 부여, 승리 횟수가 적은 팀부터 0,1,2 의 라벨을 부여
    for idx, (team, _) in enumerate(sorted(dic.items(), key= lambda x: x[1])):
        label_dic[team] = idx

    return label_dic

def homeGoal_day_mean(train, test, day):
    train[f'홈_골_{day}_평균'] = -1  # 초기값 -1로 설정
    test[f'홈_골_{day}_평균'] = -1

    teams = train['홈 팀 이름'].unique()
    for team in tqdm(teams): # train에서 고유 팀 이름을 가져오고 이를 시각적으로 표시해줌 : tqdm
        team_df = train[train['홈 팀 이름'] == team]
        # 롤링 윈도우 크기 설정
        ch_day = len(team_df) if len(team_df) < day else day # 팀의 경기 수가 주어진 day 보다 적으면, 경기 수 만큼의 윈도우 크기 사용
        idx = team_df['홈 팀 득점 수'].rolling(ch_day).mean().index.values # 롤링 윈도우 평균 계산
        val = team_df['홈 팀 득점 수'].rolling(ch_day).mean().values
        train[f'홈_골_{day}_평균'].loc[idx] = val
        test_idx = test[test['홈 팀 이름'] == team].index
        test[f'홈_골_{day}_평균'].loc[test_idx] = val[-1]
    # 결측값 처리
    train[f'홈_골_{day}_평균'] = train[f'홈_골_{day}_평균'].fillna(0)

import pandas as pd

# 빈 딕셔너리로 승점 초기화
points = {}
monthly_points = []

# 팀별 승점 계산
for index, row in win_points_df.iterrows():
    month = row['월']
    home_team = row['홈 팀 이름']
    away_team = row['원정 팀 이름']
    result = row['결과']

    # 홈팀과 원정팀의 이름이 points 딕셔너리에 없으면 추가하고 초기값 0 설정
    if home_team not in points:
        points[home_team] = 0
    if away_team not in points:
        points[away_team] = 0

    # 결과에 따라 승점 부여
    if result == '승':
        points[home_team] += 3
    elif result == '패':
        points[away_team] += 3

    # 현재 월의 승점 데이터를 저장
    monthly_points.append({
        '월': month,
        '팀 이름': home_team,
        '누적 승점': points[home_team]
    })
    monthly_points.append({
        '월': month,
        '팀 이름': away_team,
        '누적 승점': points[away_team]
    })

# 데이터프레임으로 변환 후, 월별로 팀별 누적 승점 정리
points_predict = pd.DataFrame(monthly_points).drop_duplicates(subset=['월', '팀 이름'], keep='last').sort_values(by=['월', '누적 승점'], ascending=[True, False]).reset_index(drop=True)

# 팀별 승점 계산
for index, row in win_points.iterrows():
    home_team = row['홈 팀 이름']
    away_team = row['원정 팀 이름']
    result = row['홈 팀 결과']

    # 홈팀과 원정팀의 이름이 points 딕셔너리에 없으면 추가하고 초기값 0 설정
    if home_team not in points:
        points[home_team] = 0
    if away_team not in points:
        points[away_team] = 0

    # 결과에 따라 승점 부여
    if result == '승':
        points[home_team] += 3
        points[away_team] -= 0
    elif result == '패':
        points[home_team] -= 0
        points[away_team] += 3
    elif result == '무':
        points[home_team] += 1
        points[away_team] += 1

# 승점 결과를 데이터프레임으로 변환
points_real = pd.DataFrame(list(points.items()), columns=['팀 이름', '승점']).sort_values(by='승점', ascending=False).reset_index(drop=True)
